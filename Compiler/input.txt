# VARIABLE DECLARATION (INT AND STRING)
VAR X INT 
VAR W INT IS 10
VAR Z INT IS W PLUS 10

VAR A STR
VAR B STR IS "HELLO WORLD"
VAR C STR


# ASSIGMENTS
X IS 1
A IS "CAPS "
C IS "SAYS "


# SCAN
X IS SCAN 


# PRINT
PRINT OPEN X CLOSE


# EQUAL, LESS, GREATER, NOT
PRINT OPEN X EQUAL 1 CLOSE
PRINT OPEN X LESS 1 CLOSE
PRINT OPEN X GREATER 2 CLOSE
PRINT OPEN NOT OPEN X EQUAL 0 CLOSE CLOSE


# PLUS, MINUS, DIV, TIMES
PRINT OPEN X PLUS 1 CLOSE
PRINT OPEN X MINUS 1 CLOSE
PRINT OPEN X DIV 2 CLOSE
PRINT OPEN X TIMES 5 CLOSE


# IF
IF OPEN X EQUAL 0 CLOSE DO
	PRINT OPEN X CLOSE
DOT

# IF ELSE
IF OPEN NOT OPEN X EQUAL 0 CLOSE CLOSE DO
	PRINT OPEN X CLOSE
DOT ELSE DO
	PRINT OPEN Z CLOSE
DOT


# FOR
GIVEN Y IS 0 WHILE Y LESS 5 WITH Y IS Y PLUS 1 DO
	PRINT OPEN X PLUS Y CLOSE
DOT


# STRING OPERATIONS
PRINT OPEN A JOIN C JOIN B CLOSE 	# STR JOIN STR == STR
PRINT OPEN A JOIN W CLOSE 			# STR JOIN INT == STR
PRINT OPEN Z JOIN W CLOSE 			# INT JOIN INT == STR
PRINT OPEN A GREATER B CLOSE 		# STR OPERATION STR == STR
